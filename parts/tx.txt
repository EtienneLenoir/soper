cd "CHEMIN_DU_REPO_CLONE"

$OutputFile = ".\CLI_INDO_GIT.zip"
$PartPattern = ".\parts\CLI_INDO_GIT_transfer.zip.part*"

Remove-Item $OutputFile -Force -ErrorAction SilentlyContinue

$out = [System.IO.File]::Create((Join-Path (Get-Location) $OutputFile))

try {
    Get-ChildItem $PartPattern | Sort-Object Name | ForEach-Object {
        Write-Host "Adding $($_.Name)"
        $in = [System.IO.File]::OpenRead($_.FullName)
        try {
            $in.CopyTo($out)
        }
        finally {
            $in.Close()
        }
    }
}
finally {
    $out.Close()
}

Get-FileHash $OutputFile -Algorithm SHA256
