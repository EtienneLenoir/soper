import duckdb

con = duckdb.connect("agent_memory.duckdb")

con.execute("""
CREATE TABLE IF NOT EXISTS memory (
    id INTEGER,
    kind VARCHAR,
    title VARCHAR,
    content VARCHAR,
    created_at TIMESTAMP DEFAULT now()
)
""")

def add_memory(kind, title, content):
    con.execute(
        "INSERT INTO memory VALUES ((SELECT coalesce(max(id), 0) + 1 FROM memory), ?, ?, ?, now())",
        [kind, title, content],
    )

def search_memory(query):
    return con.execute(
        """
        SELECT title, content
        FROM memory
        WHERE lower(content) LIKE '%' || lower(?) || '%'
        LIMIT 5
        """,
        [query],
    ).fetchall()

Exemple De Règles Python

def adjust_risks(row):
    flags = set(row["flags"])

    if "high_yield" in row["investment_universe"]:
        row["risk_assessment_from_text"]["credit_risk"] = "high"
        flags.add("high_yield")

    if "emerging_markets" in row["regions"]:
        flags.add("emerging_markets_exposure")
        if row["risk_assessment_from_text"]["liquidity_risk"] == "unknown":
            row["risk_assessment_from_text"]["liquidity_risk"] = "medium"

    if row["product_features"]["uses_leverage"] in ["yes", "possible"]:
        flags.add("leverage_or_possible_leverage")

    row["flags"] = sorted(flags)
    return row
